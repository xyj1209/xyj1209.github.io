猫狗队列

宠物、狗、猫的类如下，
实现一种猫狗队列的结构，要求

  • 用户可以调用add方法将cat类或者dog类实例放入队列中。
  • 用户可以调用pollAll方法，将队列中所有的实例按照进队列的先后顺序依次弹出。
  • 用户可以调用pollDog方法，将队列中的所有狗类实例按照进队列的先后顺序依次弹出。
  • 用户可以调用pollCat方法，将队列中的所有猫类实例按照进队列的先后顺序依次弹出。
  • 用户可以调用isEmpty方法，检查队列中是否有dog或cat的实例。
  • 用户可以调用isDogEmpty方法，检查队列中是否有dog的实例。
  • 用户可以调用isCatEmpty方法，检查队列中是否有cat的实例。

思路：把猫狗类封装到PetEnterQueue类中，使用一个计数器记录进入队列的顺序。

封装猫狗类，使用count先后顺序

package top.myjnxj.catdogqueue;

/**
* @ClassName PetEnterQueue
* @Description 封装猫狗类，使用count先后顺序
* @Author 江南小俊
* @Date 2018/12/22 11:21
* @Version 1.0.0
**/
public class PetEnterQueue {
    private Pet pet;
    private long count;

    public PetEnterQueue(Pet pet, long count) {
        this.pet = pet;
        this.count = count;
    }

    public Pet getPet() {
        return pet;
    }

    public long getCount() {
        return count;
    }
}

封装猫狗队列

package top.myjnxj.catdogqueue;

import java.util.LinkedList;
import java.util.Queue;

/**
* @ClassName CatDogQueue
* @Description 封装猫狗队列
* @Author 江南小俊
* @Date 2018/12/22 11:11
* @Version 1.0.0
**/
public class CatDogQueue {
    private Queue<PetEnterQueue> dogQueue;
    private Queue<PetEnterQueue> catQueue;
    private long count = 0;

    public CatDogQueue() {
        this.dogQueue = new LinkedList<>();
        this.catQueue = new LinkedList<>();
        this.count = 0;
    }

    /**
    * add方法先判断宠物类属于猫还是狗，分别构造为PetEnterQueue实例，添加到相应的队列中
    * @param pet
    */
    public void add(Pet pet) {
        if (pet instanceof Cat) {
            catQueue.add(new PetEnterQueue(pet, ++count));
        } else if (pet instanceof Dog) {
            dogQueue.add(new PetEnterQueue(pet, ++count));
        } else {

        }
    }

    /**
    * pollAll方法先判断猫和狗队列是否为空，再通过判断队首的元素的count大小，小的先出列
    * @return
    */
    public Pet pollAll() {
        if (!dogQueue.isEmpty() && !catQueue.isEmpty()) {
            if (dogQueue.peek().getCount() < catQueue.peek().getCount()) {
                return dogQueue.poll().getPet();
            } else {
                return catQueue.poll().getPet();
            }

        } else if (!dogQueue.isEmpty()) {
            return dogQueue.poll().getPet();
        } else if (!catQueue.isEmpty()) {
            return catQueue.poll().getPet();
        } else {
            return null;
        }
    }

    /**
    * 直接判断狗队列，队首出列
    * @return
    */
    public Dog pollDog() {
        if (!dogQueue.isEmpty()) {
            return (Dog) dogQueue.poll().getPet();
        } else {
            return null;
        }
    }
    /**
    * 直接判断猫队列，队首出列
    * @return
    */
    public Cat pollCat() {
        if (!catQueue.isEmpty()) {
            return (Cat) catQueue.poll().getPet();
        } else {
            return null;
        }
    }

    public boolean isEmpty() {
        return dogQueue.isEmpty() && catQueue.isEmpty();
    }

    public boolean isDogEmpty() {
        return dogQueue.isEmpty();
    }

    public boolean isCatEmpty() {
        return catQueue.isEmpty();
    }
}

快速排序

import java.util.Arrays;

/**
* Class: QuickSort.java Description: //快速排序
* 
* @author 江南小俊
* @date 2018年12月11日
* @version 1.0
*/
public class QuickSort {

    private static void swap(int[] arr, int low, int high) {
        int temp = arr[low];
        arr[low] = arr[high];
        arr[high] = temp;

    }

    public static void main(String[] args) {
        int[] arr = { 10, 3, 8, 1, 9, 6 };
        QuickSort.quickSort(arr);
        System.out.println(Arrays.toString(arr));
    }

    /**
    * @param arr
    */
    private static void quickSort(int[] arr) {
        int row = 0;
        int high = arr.length - 1;
        quickSort(arr, row, high);

    }

    /**
    * @param arr
    * @param row
    * @param high
    */
    private static void quickSort(int[] arr, int row, int high) {
        int pos;
        if (row < high) {
            pos = partition(arr, row, high);
            quickSort(arr, row, pos - 1);
            quickSort(arr, pos + 1, high);
        }

    }

    /**
    * @param arr
    * @param low
    * @param high
    * @return
    */
    private static int partition(int[] arr, int low, int high) {
        // 基准值，取最左边的值
        int key = arr[low];

        while (low < high) {
            while (low < high && arr[high] > key) {
                high--;
            }
            swap(arr, low, high);
            while (low < high && arr[low] < key) {
                low++;
            }
            swap(arr, low, high);

        }
        return high;
    }
}
